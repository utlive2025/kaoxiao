<script>

const GAS_URL = 'https://script.google.com/macros/s/AKfycbyL1WUhaBJvQqW1f5dTiFqJ-c8JHPHdPXukUroV4cJMvGona3SxOYnvI7VAkywBVEWs/exec';

let STATE = { slots:{}, winners:{}, loserW:{}, finals:{} };

let curGroup='高男組', curLevel='第一量級', curBK='高男組_第一量級';


// 核心修改：手動切換勝負邏輯

function clickW4(csk, r, mi, pid) {

  if (!STATE.winners[csk]) {

    STATE.winners[csk] = [[],[],[],[]].map((_,i)=>new Array(Math.max(1, 8>>i)).fill(null));

  }

  const W = STATE.winners[csk];


  // 如果點擊的人已經是勝者，則取消 (設為 null)；否則設為勝者

  if (W[r][mi] === pid) {

    W[r][mi] = null;

  } else {

    W[r][mi] = pid;

  }


  renderAll();     // 重新渲染畫面

  scheduleSave();  // 同步資料

}


function clickLoser(csk, mi, pid) {

  if (!STATE.loserW[csk]) STATE.loserW[csk] = [null, null, null];

  const LW = STATE.loserW[csk];

  

  if (LW[mi] === pid) {

    LW[mi] = null;

  } else {

    LW[mi] = pid;

  }

  renderAll();

  scheduleSave();

}


// 同步與渲染邏輯 (維持原本結構但調用上述 function)

async function scheduleSave(){

  setSyncStatus('syncing');

  try {

    await fetch(GAS_URL, {method:'POST', body:JSON.stringify(STATE), mode:'no-cors'});

    setSyncStatus('ok');

  } catch(e) { setSyncStatus('err'); }

}


function setSyncStatus(s){

  const dot=document.getElementById('syncDot'), lbl=document.getElementById('syncLabel');

  dot.className='sync-dot '+s;

  lbl.textContent = s==='ok'?'已同步':s==='syncing'?'同步中...':'連線失敗';

}
